# Python Data Types

## Numeric Types (`int`, `float`, `complex`)

x = 5
type(x)
# <class 'int'>

y = 5.2
type(y)
# <class 'float'>

z = 3j + 4
type(z)
# <class 'complex'>


## Text Types ('String')
a = "Hello World"
type(a)
# <class 'str'>

## Boolean Type (bool)
b = True
type(b)
# <class 'bool'>
c = 23 < 22
type(c)
# <class 'bool'>

## Sequence Type (list, tuple)
l = [1, 2, 3, 4]
type(l)
# <class 'list'>

t = ("Machine Learning", "Data Science")
type(t)
# <class 'tuple'>

## Set Type (set)
s = {"Python", "Machine Learning", "Data Science"}
type(s)
# <class 'set'>

## Dictionary Type (dict)
d = {"Name": "Brad",
     "Age": 18,
     "Adress": "Center"}  
type(d)
# <class 'dict'>
